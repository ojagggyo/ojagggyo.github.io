async function Sample(error) {
    if(error){
        throw new Error('★Error!');
    }
    else{
        return '★Sample!';
    }   
}

Sample(true)
    .then(value => {
        console.log(value);
        $("#Sample1").text(value);
    })
    .catch(err => {
        console.log(err);
        $("#Sample1").text(err);
    });
